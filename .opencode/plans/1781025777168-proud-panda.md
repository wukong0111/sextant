### 7. `result_grid.rs` — Metodos de resize (despues de `cursor_col()` en linea 518, antes del `}` de cierre del impl)

```rust
const MIN_COL_WIDTH: usize = 3;
const MAX_COL_WIDTH: usize = 80;

pub fn widen_column(&mut self) {
    let col = self.cursor_col;
    let base = self.auto_width(col);
    let current = self.col_widths.get(&col).copied().unwrap_or(base);
    self.col_widths.insert(col, (current + 2).min(Self::MAX_COL_WIDTH));
}

pub fn narrow_column(&mut self) {
    let col = self.cursor_col;
    let base = self.auto_width(col);
    let current = self.col_widths.get(&col).copied().unwrap_or(base);
    self.col_widths.insert(col, current.saturating_sub(2).max(Self::MIN_COL_WIDTH));
}

pub fn auto_fit_column(&mut self) {
    self.col_widths.remove(&self.cursor_col);
}

pub fn auto_fit_all(&mut self) {
    self.col_widths.clear();
}

fn auto_width(&self, col: usize) -> usize {
    let Some(ref result) = self.result else { return Self::MIN_COL_WIDTH };
    let Some(col_info) = result.columns.get(col) else { return Self::MIN_COL_WIDTH };
    let mut w = col_info.name.len();
    for row in &result.rows {
        if let Some(cell) = row.get(col) {
            w = w.max(cell_value_to_string(cell).len());
        }
    }
    w.clamp(Self::MIN_COL_WIDTH, 40)
}
```
